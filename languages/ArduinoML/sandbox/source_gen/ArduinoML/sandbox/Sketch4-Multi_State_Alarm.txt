
int prevOff = LOW;
int prevOn = LOW;
void setup() {
    pinMode(8, INPUT);
    pinMode(11, OUTPUT);
    pinMode(12, OUTPUT);
}

void loop() {
    if (analogicRead(8) == HIGH && prevOn == LOW) {
        analogicWrite(11, HIGH);
        analogicWrite(12, LOW);
    } else if (analogicRead(8) == LOW && prevOff == HIGH) {
        analogicWrite(11, LOW);
        analogicWrite(12, HIGH);
    }
    prevOff = analogicRead(8);
    prevOn = analogicRead(8);
}
