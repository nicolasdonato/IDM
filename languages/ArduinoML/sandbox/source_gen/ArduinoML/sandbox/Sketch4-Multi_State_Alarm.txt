
int prevOff = LOW;
int prevOn = LOW;
void setup() {
    pinMode(8, INPUT);
    pinMode(11, OUTPUT);
    pinMode(12, OUTPUT);
}

void loop() {
    if (digitalRead(8) == HIGH && prevOn == LOW) {
        digitalWrite(11, HIGH);
        digitalWrite(12, LOW);
    } else if (digitalRead(8) == LOW && prevOff == HIGH) {
        digitalWrite(11, LOW);
        digitalWrite(12, HIGH);
    }
    prevOff = digitalRead(8);
    prevOn = digitalRead(8);
}
