
void setup() {
    pinMode(12, INPUT);
    pinMode(8, OUTPUT);
    pinMode(9, OUTPUT);
}

void loop() {
    if (digitalRead(12) == LOW) {
        digitalWrite(8, LOW);
        digitalWrite(9, LOW);
    } else if (digitalRead(12) == HIGH) {
        digitalWrite(8, HIGH);
        digitalWrite(9, HIGH);
    }
}
