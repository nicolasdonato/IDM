
void setup() {
    pinMode(12, INPUT);
    pinMode(8, OUTPUT);
    pinMode(9, OUTPUT);
}

void loop() {
    if (analogicRead(12) == LOW) {
        analogicWrite(8, LOW);
        analogicWrite(9, LOW);
    } else if (analogicRead(12) == HIGH) {
        analogicWrite(8, HIGH);
        analogicWrite(9, HIGH);
    }
}
