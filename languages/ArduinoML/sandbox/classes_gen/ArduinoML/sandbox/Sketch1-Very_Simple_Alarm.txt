
void setup() {
    pinMode(12, INPUT);
    pinMode(8, OUTPUT);
    pinMode(9, OUTPUT);
}

void loop() {
    if (readAnalogic(12) == LOW) {
        writeAnalogic(LOW);
        writeAnalogic(LOW);
    } else if (readAnalogic(12) == HIGH) {
        writeAnalogic(HIGH);
        writeAnalogic(HIGH);
    }
}
