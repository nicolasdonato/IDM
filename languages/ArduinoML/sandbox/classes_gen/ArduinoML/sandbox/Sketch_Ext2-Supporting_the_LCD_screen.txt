
#include <LiquidCrystal.h>

LiquidCrystal lcd(12, 11, 5, 4, 3, 2);

void setup() {
    lcd.begin(16, 2);
    pinMode(11, INPUT);
}

void loop() {
    if (digitalRead(11) >= 52) {
        lcd.setCursor(0, 1);
        lcd.print("Fire is inside");
        lcd.setCursor(0, 2);
        lcd.print(digitalRead(11));
    } else if (digitalRead(11) < 52) {
        lcd.setCursor(0, 1);
        lcd.print("No fire");
        lcd.setCursor(0, 2);
        lcd.print(digitalRead(11));
    }
}
